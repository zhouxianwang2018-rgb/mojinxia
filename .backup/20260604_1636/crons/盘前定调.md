```python
from hermes_tools import read_file
import json, os

idx = json.loads(read_file(os.path.expanduser('~/.hermes/cron/state/moni/strategy_index.json'))['content'])
```

## Step 5: 写入主线语境到策略文件 + 激活策略

⚠️ CRITICAL: 必须同时写入 market_context 并将 status 从 pending 翻为 active。漏掉 status 翻转会导致午盘/尾盘执行全部跳过！

```python
import json, os
from datetime import datetime

TODAY = datetime.now().strftime('%Y-%m-%d')
STRATEGY_FILE = os.path.expanduser(f'~/.hermes/cron/state/moni/strategies/{TODAY}.json')

with open(STRATEGY_FILE) as f:
    strategy = json.load(f)

# 填入 Step 2-4 分析得出的结论（用实际数据替换占位符）
strategy['market_context'] = {
    'written_by': '盘前分析',
    'written_at': datetime.now().isoformat(),
    'main_themes': ['半导体', 'AI算力', '机器人'],  # ← 替换为当日的实际主线 Top3
    'sector_momentum': {},  # ← 替换为板块涨跌幅
    'external_env': '美股温和',  # ← 替换为当日外围判断
    'bias': '偏多',  # ← 替换为偏多/中性/偏空
    'risk_note': ''  # ← 替换为需要特别提示的风险
}

# ⚠️ 必须翻 status！否则午盘/尾盘全部 [SILENT]
strategy['status'] = 'active'

with open(STRATEGY_FILE, 'w') as f:
    json.dump(strategy, f, indent=2, ensure_ascii=False)

print(f"✅ status=active, market_context 已写入策略文件")
```

