### 3. 🔴 计算 risk_state（脚本输出即最终结果，绝对禁止 LLM 修改）

```bash
TODAY_PNL=<今日盈亏金额>  CURRENT_TOTAL=<当前总资产>

PARENT_CONSEC=<步骤2输出的 parent_consecutive，若为空则省略此参数>
```

🔴🔴🔴 **绝对规则：`$NEW_RISK` 的所有字段均为最终值。禁止 LLM 对 consecutive_loss_days、level、drawdown_from_peak、new_position_ban 等任何 risk_state 字段做手动修正。** 🔴🔴🔴

### 4. 🆕 创建次日策略（基于引擎排名+主线语境）

⚠️ **核心改动：scenarios 必须基于策略文件中的 `candidate_pool.ranked` + `market_context`，而非 LLM 即兴选股。**

```python
import json, datetime

d = json.load(open('$STRATEGY_FILE'))
new_risk = json.loads('''$NEW_RISK''')

# ── 读取引擎排名结果 ──
candidate_pool = d.get('candidate_pool', {})
ranked = candidate_pool.get('engine2', {}).get('ranked', [])
market_ctx = d.get('market_context', {})
main_themes = market_ctx.get('main_themes', [])
bias = market_ctx.get('bias', '中性')

# ── 按优先级取候选 ──
# P0(权重≥5): 双引擎+主线 — 最高优先级
# P1(权重≥3): 引擎一满分 或 引擎二Top5+主线
# P2(权重≥1): 其他
priority_map = {r['code']: r for r in ranked}
p0 = [r for r in ranked if r['priority'] >= 5]
p1 = [r for r in ranked if 3 <= r['priority'] < 5]
p2 = [r for r in ranked if 1 <= r['priority'] < 3]

print(f"候选池: P0={len(p0)} P1={len(p1)} P2={len(p2)}")
print(f"主线: {main_themes}")
print(f"偏置: {bias}")

# ── 构建 scenarios ──
scenarios = []

# 场景1: 顺势建仓（优先P0+P1）
if p0 or p1:
    top = (p0 + p1)[:3]
    codes = ', '.join(f"{r['code']}{r['name']}" for r in top)
    reasons = ', '.join(r['reason'] for r in top)
    scenarios.append({
        'if': f"大盘开盘平稳（上证开盘点±0.5%以内），{main_themes[0] if main_themes else '主线'}板块延续强势",
        'then': f"优先建仓候选: {codes}。单票≤¥80,000，分2-3笔。理由: {reasons}",
        'type': 'momentum_build'
    })

# 场景2: 震荡观望
scenarios.append({
    'if': '大盘平开窄幅震荡，无明确方向',
    'then': '等待10:00后确认方向再建仓。观察候选池中标的的量价配合',
    'type': 'wait_confirm'
})

# 场景3: 防御
scenarios.append({
    'if': '大盘低开>1%或外围大跌',
    'then': '延迟建仓至午后。维持现金为主。不在恐慌中进场',
    'type': 'defensive'
})

# 场景4: 止损
scenarios.append({
    'if': '单票持仓浮亏>5%',
    'then': '无条件止损，市价卖出全部该股持仓',
    'type': 'stop_loss'
})

# ── 构建次日策略 ──
mode_map = {'normal': 'normal', 'cautious': 'conservative', 'defensive': 'defensive', 'emergency': 'emergency'}
mode = mode_map.get(new_risk['level'], 'conservative')

max_buy = 0 if new_risk['new_position_ban'] else (
    400000 if new_risk['level'] == 'normal' else 200000
)

thesis = f"{'✅ 风险正常，可以建仓' if new_risk['level']=='normal' else '⚠️ 保守模式，控制仓位'}。"
thesis += f"主线: {', '.join(main_themes[:3]) if main_themes else '待盘前分析确认'}。"
thesis += f"候选池: {len(ranked)}只(P0:{len(p0)} P1:{len(p1)})。"

s = {
    'trading_day': '$NEXT_DAY',
    'parent_strategy': '$TODAY',
    'status': 'pending',
    'created_by': '收盘复盘',
    'created_at': datetime.datetime.now().isoformat(),
    'valid_until': '$NEXT_DAY' + 'T15:00:00+08:00',
    'closed_at': None,
    'risk_state': new_risk,
    'current_strategy': {
        'id': '复盘-' + d['current_strategy']['id'],
        'mode': mode,
        'core_thesis': thesis,
        'max_total_buy_today': max_buy,
        'max_single_trade_pct': new_risk['hard_single_stock_cap'],
        'per_stock': {},
        'scenarios': scenarios,
        'valid_until': '$NEXT_DAY' + 'T15:00:00+08:00'
    },
    'overrides': [],
    'execution_log': []
}

json.dump(s, open('/tmp/moni_next.json', 'w'), indent=2, ensure_ascii=False)
print(f"✅ 次日策略已生成: {len(scenarios)} scenarios, {len(ranked)} 候选")
```
